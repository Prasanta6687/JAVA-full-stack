#include<stdio.h>
#include<conio.h>
#include<graphics.h>
void main()
{
inti,gd=DETECT,gm,x,y,red=1;
initgraph(&gd,&gm,"C:\\TURBOC3\\BGI");
x=getmaxx()/2;
y=getmaxy()/2;
for(i=1;i<=11;i++)
{
 setfillstyle(i,i);
 red=20*i;
 circle(x,y,red);
 floodfill(x+(red-2),y,getmaxcolor());
 sound(200);
 delay(100);
 nosound();
 dealy(100);
 }
getch()
}
